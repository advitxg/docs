```python python
import requests

headers = {
  'content-type': 'application/json'
}

data = [{
  "domestic_blockchain": "ethereum",
  "domestic_token": "0xa0b86991c6218b36c1d19d4a2e9eb0ce3606eb48",
  "foreign_blockchain": "ethereum",
```
